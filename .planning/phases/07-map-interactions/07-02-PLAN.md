    **Pure opacity functions** (inline, mirroring map.mjs exports — map.mjs is Node-testable; the inline script cannot import it via bare specifier):
    ```javascript
    function tagOpacity(doc) {
      if (selectedTags.size === 0) return 1.0;
      const match = anyAllMode === 'ANY'
        ? doc.tags.some(t => selectedTags.has(t))
        : [...selectedTags].every(t => doc.tags.includes(t));
      return match ? 1.0 : DIM_OPACITY;
    }
    function searchOpacity(doc) {
      if (matchingDocIds === null) return 1.0;
      return matchingDocIds.has(doc.id) ? 1.0 : DIM_OPACITY;
    }
    function gaussianOp(doc) {
      if (!timelineDate || !doc.date) return 1.0;
      const docMs = new Date(doc.date).getTime();
      const selMs = timelineDate.getTime();
      const deltaDays = Math.abs(docMs - selMs) / 86400000;
      return Math.exp(-(deltaDays * deltaDays) / (2 * SIGMA_DAYS * SIGMA_DAYS));
    }
    function composedOp(doc) {
      return Math.min(tagOpacity(doc), searchOpacity(doc), gaussianOp(doc));
    }
    ```

    **kNearest inline** (map.mjs is not importable in is:inline; duplicate the logic):
    ```javascript
    function kNearest(dot, k) {
      return dotPositions
        .filter(d => d !== dot)
        .map(d => ({ d, dist: Math.sqrt((d.cx - dot.cx) ** 2 + (d.cy - dot.cy) ** 2) }))
        .sort((a, b) => a.dist - b.dist)
        .slice(0, k)
        .map(e => e.d);
    }
    ```

    **drawDot helper:**
    ```javascript
    function drawDot(cx, cy, color, opacity) {
      ctx.save();
      ctx.globalAlpha = opacity;
      ctx.beginPath();
      ctx.arc(cx, cy, 5, 0, 2 * Math.PI);
      ctx.fillStyle = color;
      ctx.shadowColor = color;
      ctx.shadowBlur = opacity < 0.5 ? 4 : 8; // reduce glow for dimmed dots (pitfall 2)
      ctx.fill();
      ctx.restore();
    }
    ```

    **drawNeighborLines:**
    ```javascript
    function drawNeighborLines(dot) {
      const neighbors = kNearest(dot, NEIGHBOR_COUNT);
      ctx.save();
      ctx.strokeStyle = '#8888AA';
      ctx.lineWidth = 1;
      ctx.globalAlpha = 0.4;
      for (const n of neighbors) {
        ctx.beginPath();
        ctx.moveTo(dot.cx, dot.cy);
        ctx.lineTo(n.cx, n.cy);
        ctx.stroke();
      }
      ctx.restore();
    }
    ```

    **Central redraw() function:**
    ```javascript
    function redraw() {
      ctx.clearRect(0, 0, W, H);
      if (selectedDot) drawNeighborLines(selectedDot);
      for (const dot of dotPositions) {
        const opacity = composedOp(dot.doc);
        const color = CATEGORY_COLORS[dot.doc.category] || FALLBACK_COLOR;
        drawDot(dot.cx, dot.cy, color, opacity);
      }
    }
    ```

    **JS addition** in the inline script — add openPanel, closePanel, and click handler after the redraw() definition:
    ```javascript
    function openPanel(doc) {
      const panel = document.getElementById('map-panel');
      const content = document.getElementById('map-panel-content');
      const tagPills = doc.tags.map(t => `<span class="tag-pill">${t}</span>`).join(' ');
      content.innerHTML = `
        <h2 class="map-panel-title">${doc.title}</h2>
        <div class="map-panel-tags">${tagPills}</div>
        <p class="map-panel-summary">${doc.summary || '<em>No summary available.</em>'}</p>
        <a href="/${doc.category}/${doc.id}/" class="map-panel-link">Open article →</a>
      `;
      panel.classList.add('open');
    }
    function closePanel() {
      document.getElementById('map-panel').classList.remove('open');
    }

    // Click: select dot, draw neighbors, open panel; second click on same dot = deselect
    canvas.addEventListener('click', (event) => {
      const r = canvas.getBoundingClientRect();
      const mx = event.clientX - r.left, my = event.clientY - r.top;
      const hit = hitTest(mx, my, dotPositions, HIT_RADIUS);
      if (!hit) { return; }
      if (selectedDot === hit) {
        selectedDot = null;
        closePanel();
      } else {
        selectedDot = hit;
        openPanel(hit.doc);
      }
      redraw();
    });

    document.getElementById('map-panel-close').addEventListener('click', () => {
      selectedDot = null;
      closePanel();
      redraw();
    });
    ```
