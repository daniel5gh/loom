    **B. Extend the Base.astro bundled `<script>`** with vim navigation.
    Add the following to the existing script (the `overlayOpen`, `isTypingContext`, and global keydown listener already exist from plan 03 — extend, do not duplicate):

    ```javascript
    // Vim navigation state
    let lastGKey = 0;
    const modeLabel = document.getElementById('vim-mode-label');
    const contextLabel = document.getElementById('vim-context-label');

    function updateStatusBar(mode, context) {
      if (modeLabel) modeLabel.textContent = mode;
      if (contextLabel) contextLabel.textContent = context || '';
    }

    function selectVimItem(items, newIndex) {
      items.forEach(el => el.classList.remove('vim-selected'));
      if (newIndex < 0 || newIndex >= items.length) return;
      items[newIndex].classList.add('vim-selected');
      items[newIndex].scrollIntoView({ block: 'nearest' });
      updateStatusBar('NORMAL', `${newIndex + 1}/${items.length}`);
    }

    function getVimItems() {
      const list = document.querySelector('[data-vim-list]');
      if (!list) return [];
      return [...list.querySelectorAll('[data-vim-item]')];
    }

    function currentVimIndex(items) {
      return items.findIndex(el => el.classList.contains('vim-selected'));
    }

    function handleVimKey(e) {
      // Only act on list pages; no-op on canvas pages (graph, map)
      const items = getVimItems();
      if (!items.length) return;

      let idx = currentVimIndex(items);
      if (idx === -1) idx = 0;

      if (e.key === 'g') {
        const now = Date.now();
        if (now - lastGKey < 500) {
          // gg — go to top
          selectVimItem(items, 0);
          lastGKey = 0;
        } else {
          lastGKey = now;
        }
        return;
      }

      // Any key other than 'g' resets the gg timer
      lastGKey = 0;

      if (e.key === 'j') {
        e.preventDefault();
        selectVimItem(items, Math.min(idx + 1, items.length - 1));
      } else if (e.key === 'k') {
        e.preventDefault();
        selectVimItem(items, Math.max(idx - 1, 0));
      } else if (e.key === 'G') {
        e.preventDefault();
        selectVimItem(items, items.length - 1);
      } else if (e.key === 'Enter') {
        const selectedItem = items.find(el => el.classList.contains('vim-selected'));
        const link = selectedItem?.querySelector('a');
        if (link) {
          window.location.href = link.href;
        }
      }
    }

    // Extend global keydown handler — add vim handling branch
    // (The existing handler from plan 03 fires for '/' and 'Escape'.
    //  Add vim key handling for when overlay is closed and not typing.)
    document.addEventListener('keydown', (e) => {
      // If overlay is open, skip vim keys (overlay has its own keydown on input)
      if (overlayOpen) return;
      if (isTypingContext()) return;
      handleVimKey(e);
    });

    // Initialize status bar on page load for list pages
    document.addEventListener('DOMContentLoaded', () => {
      const items = getVimItems();
      if (items.length > 0) {
        updateStatusBar('NORMAL', `0/${items.length}`);
      }
    });
    ```

    IMPORTANT: The global keydown listener from plan 03 already handles `/` and `Escape`. The vim extension is a SECOND `addEventListener('keydown')` — this is fine. Do not remove the plan 03 handler. Alternatively, if the executor prefers, consolidate both into a single listener — either approach works as long as all keys are handled.
