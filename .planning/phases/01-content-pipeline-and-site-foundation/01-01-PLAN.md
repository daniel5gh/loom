import re
files = {
    'ai-tools-and-services/airllm.md': ['ai', 'llm', 'inference', 'optimization', 'open-source'],
    'ai-tools-and-services/langflow.md': ['ai', 'orchestration', 'workflow', 'llm', 'open-source'],
    'ai-tools-and-services/lovable.md': ['ai', 'app-builder', 'website-builder', 'no-code', 'chat', 'saas'],
    'ai-tools-and-services/pageindex.md': ['ai', 'rag', 'retrieval', 'llm', 'document-analysis', 'open-source'],
}
for path, expected_tags in files.items():
    with open(path) as f:
        content = f.read()
    tags_match = re.search(r'^tags:\s*\[(.+)\]', content, re.MULTILINE)
    assert tags_match, f'{path}: no tags line found'
    tags = [t.strip() for t in tags_match.group(1).split(',')]
    for tag in tags:
        assert tag == tag.lower(), f'{path}: uppercase tag found: {tag}'
        assert ' ' not in tag, f'{path}: space in tag: {tag}'
    assert tags == expected_tags, f'{path}: tags mismatch. Got {tags}, expected {expected_tags}'
    print(f'OK: {path} tags = {tags}')
print('All tag normalization verified.')
" 2>&1
