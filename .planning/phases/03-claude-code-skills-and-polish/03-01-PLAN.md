    After the existing tag link check (~line 74), add a new check:
    ```js
    // Spot-check a document page for Shiki-generated markup (REQ-016)
    // Shiki produces <pre class="astro-code"> wrappers
    let shikiCheckPassed = false;
    for (const cat of CATEGORY_DIRS) {
      const catPath = join(DIST, cat);
      if (!existsSync(catPath)) continue;
      const entries = readdirSync(catPath, { withFileTypes: true }).filter(e => e.isDirectory());
      for (const entry of entries) {
        const htmlFile = join(catPath, entry.name, 'index.html');
        if (!existsSync(htmlFile)) continue;
        const html = readFileSync(htmlFile, 'utf8');
        if (html.includes('astro-code')) {
          console.log(`OK: Shiki astro-code markup found in dist/${cat}/${entry.name}/index.html`);
          shikiCheckPassed = true;
          passed++;
          break;
        }
      }
      if (shikiCheckPassed) break;
    }
    if (!shikiCheckPassed) {
      console.log('WARN: No astro-code markup found — no document contains a fenced code block, or Shiki not enabled');
    }
    ```
    Note: This is a warn (not error) because some documents may have no code blocks. The check confirms Shiki is active when code blocks exist.
