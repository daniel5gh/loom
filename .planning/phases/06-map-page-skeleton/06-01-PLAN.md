  <name>Task 2: Create src/pages/map.astro — data island, HiDPI canvas, dots, tooltip</name>
  <files>src/pages/map.astro</files>
  <action>
    Create src/pages/map.astro. Mirror graph.astro structure exactly: frontmatter imports data, serializes to JSON string, passes via data island script element, all canvas/window code inside is:inline type="module".

    FRONTMATTER (runs at build time — no browser globals here):
    ```astro
    ---
    import Base from '../layouts/Base.astro';
    import embeddingsData from '../data/embeddings.json';
    const mapData = JSON.stringify(embeddingsData);
    ---
    ```

    HTML BODY:
    - Wrap in Base title="Map"
    - A div.map-page containing div#map-canvas-wrap.map-canvas-wrap
    - canvas#map-canvas inside the wrap
    - div#map-tooltip.map-tooltip inside the wrap (hidden by default)
    - script type="application/json" id="map-data" set:html={mapData} (data island)
    - script is:inline type="module" (all canvas code)

    INLINE SCRIPT — place after all HTML so getBoundingClientRect() has layout:

    1. Parse data island: const data = JSON.parse(document.getElementById('map-data').textContent); const docs = data.documents;

    2. Category color map (reuse graph.astro's three neon colors):
       const CATEGORY_COLORS = { 'ai-tools-and-services': '#00E5FF', 'cloud-ai-platforms': '#FF2D78', 'companies': '#39FF14' };
       const FALLBACK_COLOR = '#8888AA';

    3. HiDPI canvas setup (do NOT hardcode dimensions — read from getBoundingClientRect):
       const canvas = document.getElementById('map-canvas');
       const dpr = window.devicePixelRatio || 1;
       const rect = canvas.getBoundingClientRect();
       canvas.width = rect.width * dpr;
       canvas.height = rect.height * dpr;
       canvas.style.width = rect.width + 'px';
       canvas.style.height = rect.height + 'px';
       const ctx = canvas.getContext('2d');
       ctx.scale(dpr, dpr);
       const W = rect.width, H = rect.height;

    4. Coordinate normalization (guard single-doc edge case):
       const xs = docs.map(d => d.x), ys = docs.map(d => d.y);
       const xMin = Math.min(...xs), xMax = Math.max(...xs);
       const yMin = Math.min(...ys), yMax = Math.max(...ys);
       const PAD = 40;
       const scaleX = xMax === xMin ? () => W/2 : v => PAD + (v - xMin)/(xMax - xMin)*(W - 2*PAD);
       const scaleY = yMax === yMin ? () => H/2 : v => PAD + (v - yMin)/(yMax - yMin)*(H - 2*PAD);

    5. Build dotPositions array and draw dots (neon glow via shadowBlur):
       const HIT_RADIUS = 8;
       const dotPositions = [];
       ctx.clearRect(0, 0, W, H);
       for (const doc of docs) {
         const cx = scaleX(doc.x), cy = scaleY(doc.y);
         dotPositions.push({ doc, cx, cy });
         const color = CATEGORY_COLORS[doc.category] || FALLBACK_COLOR;
         ctx.save();
         ctx.beginPath();
         ctx.arc(cx, cy, 5, 0, 2 * Math.PI);
         ctx.fillStyle = color;
         ctx.shadowColor = color;
         ctx.shadowBlur = 8;
         ctx.fill();
         ctx.restore();
       }

    6. Tooltip logic:
       const tooltip = document.getElementById('map-tooltip');
       canvas.addEventListener('mousemove', (event) => {
         const r = canvas.getBoundingClientRect();
         const mx = event.clientX - r.left, my = event.clientY - r.top;
         let hit = null;
         for (const dot of dotPositions) {
           const dx = dot.cx - mx, dy = dot.cy - my;
           if (Math.sqrt(dx*dx + dy*dy) < HIT_RADIUS) { hit = dot; break; }
         }
         if (hit) {
           tooltip.style.display = 'block';
           tooltip.style.left = (mx + 12) + 'px';
           tooltip.style.top = (my + 12) + 'px';
           tooltip.innerHTML = '<strong>' + hit.doc.title + '</strong><br>' + hit.doc.tags.join(', ');
         } else {
           tooltip.style.display = 'none';
         }
       });
       canvas.addEventListener('mouseleave', () => { tooltip.style.display = 'none'; });
  </action>
