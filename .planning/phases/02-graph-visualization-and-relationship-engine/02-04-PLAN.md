    Open src/pages/graph.astro and add the three interaction behaviors inside the existing client script. Insert them AFTER the simulation and node/link setup but BEFORE the tick handler (so linkedByIndex is built before any mutation).

    === 1. Build adjacency lookup (insert before simulation line) ===
    ```javascript
    // Build adjacency lookup BEFORE simulation mutates source/target to node objects
    const linkedByIndex = {};
    edges.forEach(d => {
      linkedByIndex[`${d.source},${d.target}`] = true;
      linkedByIndex[`${d.target},${d.source}`] = true;
    });

    function isNeighbor(a, b) {
      return a.id === b.id || linkedByIndex[`${a.id},${b.id}`];
    }
    ```

    === 2. Add node interaction handlers (after node.append('text'), before tick handler) ===
    ```javascript
    node
      .on('mouseover', (event, d) => {
        node.style('opacity', n => isNeighbor(d, n) ? 1 : 0.15);
        link.style('opacity', l =>
          (l.source.id === d.id || l.target.id === d.id) ? 1 : 0.05);
      })
      .on('mouseout', () => {
        node.style('opacity', 1);
        link.style('opacity', 1);
      })
      .on('click', (event, d) => {
        window.location.href = d.url;
      });
    ```

    Note: After simulation runs, l.source and l.target are node objects (not strings), so use l.source.id and l.target.id.

    === 3. Add tag filter interaction (after tick handler, at end of script) ===
    ```javascript
    // Tag filter sidebar interaction
    let activeTag = null;

    d3.selectAll('.tag-filter-btn').on('click', function() {
      const tag = this.dataset.tag;
      if (activeTag === tag) {
        // Toggle off — restore all
        activeTag = null;
        node.style('opacity', 1);
        link.style('opacity', 1);
        d3.selectAll('.tag-filter-btn').classed('active', false);
      } else {
        activeTag = tag;
        node.style('opacity', n => n.tags.includes(tag) ? 1 : 0.1);
        link.style('opacity', l =>
          l.source.tags.includes(tag) && l.target.tags.includes(tag) ? 1 : 0.05);
        d3.selectAll('.tag-filter-btn').classed('active', false);
        d3.select(this).classed('active', true);
      }
    });
    ```

    Note: d3.selectAll('.tag-filter-btn') selects DOM elements rendered by Astro in the template section. This works because the script runs after DOM is loaded (it's at the bottom of the page in the module script). Use `function()` not arrow function so `this` refers to the clicked button.
